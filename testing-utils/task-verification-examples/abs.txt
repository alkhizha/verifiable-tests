; using IF theorem we get

; Q => BB
(IMPLIES
  (EQ x X)
  (OR
    (< x 0)
    (>= x 0)
  )
)

; Q ^ B1 => wp(S1, R)
; where B1 - x < 0, S1 - x := -x 
(IMPLIES
  (AND (EQ x X) (< x 0))
  (OR
    (AND (EQ (- 0 x) X) (<= 0 X))
    (AND (EQ (- 0 x) (- 0 X)) (<= X 0))
  )
)

; Q ^ B2 => wp(S2, R)
; where B1 - x >= 0, S2 - skip
(IMPLIES
  (AND (EQ x X) (>= x 0))
  (OR
    (AND (EQ x X) (<= 0 X))
    (AND (EQ x (- 0 X)) (<= X 0))
  )
)
